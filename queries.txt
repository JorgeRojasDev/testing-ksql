 1  CREATE STREAM INST_STREAM (ROWKEY VARCHAR, FECHA_REVREM VARCHAR, DATOS_ADICIONALES VARCHAR) WITH(KAFKA_TOPIC = 'inst.stream', VALUE_FORMAT = 'JSON');

 2  CREATE TABLE INST_TABLE WITH (KAFKA_TOPIC='INST_TABLE', PARTITIONS=1, REPLICAS=1) AS SELECT
    INST_STREAM.ROWKEY ROWKEY,
    LATEST_BY_OFFSET(INST_STREAM.FECHA_REVREM) FECHA_REVREM,
    LATEST_BY_OFFSET(INST_STREAM.DATOS_ADICIONALES) DATOS_ADICIONALES
    FROM INST_STREAM INST_STREAM
    GROUP BY INST_STREAM.ROWKEY
    EMIT CHANGES;
